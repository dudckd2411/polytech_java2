public class Main {
    public static void main(String[] args) {
        printWelcomeMessage();

        int result = add(10, 20);
        System.out.println("10 + 20 = " + result);

        boolean adult = isAdult(20);
        System.out.println("성인 여부: " + adult);
    }

    static void printWelcomeMessage() {
        System.out.println("Java 함수 연습");
    }

    static int add(int a, int b) {
        return a + b;
    }

    static boolean isAdult(int age) {
        return age >= 19;
    }
}

package com.survivalcoding;

public class Cleric {
final int MAX_HP = 50;
final int MAX_MP = 10;
final int selfAidMpCost = 5; // 매직넘버를 없애기 위해 상수 설정
String name;
int hp = MAX_HP;
int mp = MAX_MP;

    void selfAid() {
        if (mp < selfAidMpCost) {
            System.out.println("MP가 부족합니다.");
            return;
        }

        mp -= selfAidMpCost;
        hp = MAX_HP;

        System.out.println("셀프 에이드");
    }

    int pray(int sec) {
        int beforeMp = this.mp;

        this.mp += sec + new java.util.Random().nextInt(3);

        if (this.mp > this.MAX_MP) {
            this.mp = this.MAX_MP;
        }

        return this.mp - beforeMp;

    }

}
